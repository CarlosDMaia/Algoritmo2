#include <stdio.h>
#include <stdlib.h>

int somasequencial (int n){
    int i, soma = 0;
    for(i = 0; i <= n; i++){
        soma = soma + i;
    }
        return soma;
}

int main()
{
    int n;
    int resultado;
    printf("Digite quantos numeros sequenciais quer somar: \n");
    scanf("%d", &n);
    resultado = somasequencial(n);
    printf("A resultado e: %d", resultado);
    return 0;
}
